tar zxvf <directory_of_tarball_file> -C ./
rm <directory_of_tarball_file>
cp ./opentabletdriver/etc/udev/rules.d /etc/udev
rm /usr/local/bin
rm /usr/local/lib
rm /usr/local/share
cp ./opentabletdriver/usr/local/bin /usr/local
cp ./opentabletdriver/usr/local/lib /usr/local
cp ./opentabletdriver/usr/local/share /usr/local
rm ./opentabletdriver
